### 3-2. 테이블 생성 (SQL)

```sql
-- 팀원 테이블
CREATE TABLE team_members (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(100),
    team VARCHAR(50), -- 'sales', 'cleaning', 'data', 'customer_support'
    slack_id VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 태스크 테이블
CREATE TABLE tasks (
    id SERIAL PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    status VARCHAR(50) NOT NULL, -- 'PENDING', 'IN_PROGRESS', 'COMPLETED'
    team VARCHAR(50) NOT NULL,
    assigned_to INT REFERENCES team_members(id),
    created_by INT REFERENCES team_members(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    due_date TIMESTAMP,
    
    -- 영업팀용
    customer_name VARCHAR(100),
    property_location VARCHAR(255),
    contract_stage VARCHAR(50), -- 'CONTACTED', 'VISITED', 'CONTRACT_IN_PROGRESS', 'CONTRACT_SIGNED'
    
    -- 청소팀용
    cleaning_type VARCHAR(50), -- 'SETUP', 'REGULAR', 'DEEP_CLEAN'
    
    -- 연관된 숙소 (HIERO 연동)
    property_id INT,
    
    -- 부모 태스크 (태스크 체인)
    parent_task_id INT REFERENCES tasks(id)
);

-- 태스크 히스토리 (감사 추적)
CREATE TABLE task_history (
    id SERIAL PRIMARY KEY,
    task_id INT NOT NULL REFERENCES tasks(id),
    old_status VARCHAR(50),
    new_status VARCHAR(50),
    changed_by INT REFERENCES team_members(id),
    changed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    note TEXT
);

-- 서브태스크 (체크리스트)
CREATE TABLE task_subtasks (
    id SERIAL PRIMARY KEY,
    task_id INT NOT NULL REFERENCES tasks(id),
    title VARCHAR(255),
    is_completed BOOLEAN DEFAULT FALSE,
    order_index INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 댓글
CREATE TABLE task_comments (
    id SERIAL PRIMARY KEY,
    task_id INT NOT NULL REFERENCES tasks(id),
    author_id INT REFERENCES team_members(id),
    content TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 알림
CREATE TABLE notifications (
    id SERIAL PRIMARY KEY,
    recipient_id INT REFERENCES team_members(id),
    task_id INT REFERENCES tasks(id),
    type VARCHAR(50), -- 'TASK_CREATED', 'TASK_UPDATED', 'COMMENT', 'ASSIGNED'
    message TEXT,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 워크플로우 자동화 규칙
CREATE TABLE workflows (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255),
    trigger_status VARCHAR(50), -- 어떤 상태 변경 시 트리거
    trigger_team VARCHAR(50),   -- 어느 팀에서 트리거
    action_type VARCHAR(50),     -- 'CREATE_TASK', 'SEND_MESSAGE', 'UPDATE_STATUS'
    action_team VARCHAR(50),     -- 어느 팀에 액션할지
    action_config JSON,          -- 액션 상세 설정
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 인덱스 추가 (성능)
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_team ON tasks(team);
CREATE INDEX idx_tasks_assigned_to ON tasks(assigned_to);
CREATE INDEX idx_task_history_task_id ON task_history(task_id);
CREATE INDEX idx_notifications_recipient ON notifications(recipient_id);
```

## Step 5: 핵심 API 구현

### 5-1. src/routes/tasks.js

```javascript
// src/routes/tasks.js
const express = require('express');
const db = require('../database');
const { triggerWorkflow } = require('../utils/automation');

const router = express.Router();

// 1️⃣ 모든 태스크 조회 (팀별)
router.get('/', async (req, res) => {
    try {
        const { team, status } = req.query;
        let query = `
            SELECT t.*, tm.name as assigned_name
            FROM tasks t
            LEFT JOIN team_members tm ON t.assigned_to = tm.id
            WHERE 1=1
        `;
        const params = [];

        if (team) {
            query += ` AND t.team = $${params.length + 1}`;
            params.push(team);
        }
        if (status) {
            query += ` AND t.status = $${params.length + 1}`;
            params.push(status);
        }

        query += ` ORDER BY t.created_at DESC`;
        const result = await db.query(query, params);
        res.json(result.rows);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

// 2️⃣ 단일 태스크 조회
router.get('/:id', async (req, res) => {
    try {
        const result = await db.query(
            `SELECT * FROM tasks WHERE id = $1`,
            [req.params.id]
        );
        if (result.rows.length === 0) {
            return res.status(404).json({ error: 'Task not found' });
        }

        // 댓글도 함께 조회
        const comments = await db.query(
            `SELECT tc.*, tm.name FROM task_comments tc
             LEFT JOIN team_members tm ON tc.author_id = tm.id
             WHERE task_id = $1 ORDER BY created_at DESC`,
            [req.params.id]
        );

        // 서브태스크 조회
        const subtasks = await db.query(
            `SELECT * FROM task_subtasks WHERE task_id = $1 ORDER BY order_index`,
            [req.params.id]
        );

        res.json({
            ...result.rows[0],
            comments: comments.rows,
            subtasks: subtasks.rows,
        });
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

// 3️⃣ 새 태스크 생성
router.post('/', async (req, res) => {
    try {
        const {
            title,
            description,
            team,
            assigned_to,
            created_by,
            customer_name,
            property_location,
            contract_stage,
            subtasks = [],
        } = req.body;

        // 태스크 삽입
        const result = await db.query(
            `INSERT INTO tasks (
                title, description, team, assigned_to, created_by,
                status, customer_name, property_location, contract_stage
            ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)
            RETURNING *`,
            [
                title,
                description,
                team,
                assigned_to,
                created_by,
                'PENDING',
                customer_name,
                property_location,
                contract_stage,
            ]
        );

        const taskId = result.rows[0].id;

        // 서브태스크 추가
        for (let i = 0; i < subtasks.length; i++) {
            await db.query(
                `INSERT INTO task_subtasks (task_id, title, order_index)
                 VALUES ($1, $2, $3)`,
                [taskId, subtasks[i], i]
            );
        }

        res.status(201).json(result.rows[0]);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

// 4️⃣ 태스크 상태 변경 (드래그앤드롭)
router.patch('/:id/status', async (req, res) => {
    try {
        const { status, changed_by } = req.body;
        const taskId = req.params.id;

        // 기존 상태 조회
        const oldTask = await db.query(
            `SELECT * FROM tasks WHERE id = $1`,
            [taskId]
        );
        const oldStatus = oldTask.rows[0].status;

        // 상태 업데이트
        const result = await db.query(
            `UPDATE tasks SET status = $1, updated_at = NOW()
             WHERE id = $2 RETURNING *`,
            [status, taskId]
        );

        // 히스토리 기록
        await db.query(
            `INSERT INTO task_history (task_id, old_status, new_status, changed_by)
             VALUES ($1, $2, $3, $4)`,
            [taskId, oldStatus, status, changed_by]
        );

        // 🔥 자동화 트리거 (중요!)
        await triggerWorkflow(taskId, oldTask.rows[0], status);

        res.json(result.rows[0]);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

// 5️⃣ 태스크 삭제
router.delete('/:id', async (req, res) => {
    try {
        await db.query(`DELETE FROM task_comments WHERE task_id = $1`, [
            req.params.id,
        ]);
        await db.query(`DELETE FROM task_subtasks WHERE task_id = $1`, [
            req.params.id,
        ]);
        const result = await db.query(
            `DELETE FROM tasks WHERE id = $1 RETURNING *`,
            [req.params.id]
        );
        res.json(result.rows[0]);
    } catch (err) {
        res.status(500).json({ error: err.message });
    }
});

module.exports = router;
```

### 5-2. src/utils/automation.js (자동화 핵심!)

```javascript
// src/utils/automation.js
const db = require('../database');
const { sendSlackMessage } = require('./slack');

/**
 * 워크플로우 자동화 트리거
 * 
 * 예: 영업팀이 CONTRACT_SIGNED → 자동으로 청소팀에 태스크 생성
 */
async function triggerWorkflow(taskId, task, newStatus) {
    try {
        console.log(`🔄 Triggering workflow for task ${taskId}: ${newStatus}`);

        // 1. 해당하는 워크플로우 규칙 찾기
        const workflows = await db.query(
            `SELECT * FROM workflows 
             WHERE is_active = true 
             AND trigger_team = $1 
             AND trigger_status = $2`,
            [task.team, newStatus]
        );

        // 2. 각 워크플로우 실행
        for (const workflow of workflows.rows) {
            console.log(`✅ Executing workflow: ${workflow.name}`);

            if (workflow.action_type === 'CREATE_TASK') {
                // 새 태스크 자동 생성
                await createTaskAutomatically(
                    workflow.action_config,
                    taskId,
                    task
                );
            } else if (workflow.action_type === 'SEND_MESSAGE') {
                // 메시지 발송
                await sendSlackMessage(
                    workflow.action_config.channel,
                    workflow.action_config.message
                );
            }
        }
    } catch (err) {
        console.error('❌ Workflow trigger error:', err);
    }
}

/**
 * 자동으로 태스크 생성 (가장 중요한 자동화)
 */
async function createTaskAutomatically(config, parentTaskId, parentTask) {
    try {
        const result = await db.query(
            `INSERT INTO tasks (
                title, description, team, status, parent_task_id,
                customer_name, property_location
            ) VALUES ($1, $2, $3, $4, $5, $6, $7)
            RETURNING *`,
            [
                config.title, // e.g., "초기 세팅"
                config.description,
                config.team, // e.g., "cleaning"
                'PENDING',
                parentTaskId,
                parentTask.customer_name,
                parentTask.property_location,
            ]
        );

        console.log(`✨ New task created: ${result.rows[0].id}`);

        // Slack 알림 발송
        await sendSlackMessage(
            `#${config.team}`,
            `🆕 새로운 태스크: ${config.title}\n주소: ${parentTask.property_location}`
        );

        return result.rows[0];
    } catch (err) {
        console.error('❌ Error creating task:', err);
    }
}

module.exports = {
    triggerWorkflow,
    createTaskAutomatically,
};
```

```javascript
// src/store/taskStore.js
import create from 'zustand';
import axios from 'axios';

const API = axios.create({
    baseURL: 'http://localhost:3001/api',
});

const useTaskStore = create((set, get) => ({
    tasks: [],
    notifications: [],
    
    // 태스크 조회
    fetchTasks: async (team, status) => {
        try {
            const response = await API.get('/tasks', {
                params: { team, status }
            });
            set({ tasks: response.data });
        } catch (err) {
            console.error('Failed to fetch tasks:', err);
        }
    },

    // 태스크 상태 변경
    updateTaskStatus: async (taskId, newStatus) => {
        try {
            const response = await API.patch(`/tasks/${taskId}/status`, {
                status: newStatus,
                changed_by: 1, // 현재 사용자 ID (나중에 auth로 변경)
            });
            
            // UI 즉시 업데이트
            set((state) => ({
                tasks: state.tasks.map((t) =>
                    t.id === taskId ? response.data : t
                ),
            }));
            
            return response.data;
        } catch (err) {
            console.error('Failed to update task:', err);
        }
    },

    // 새 태스크 추가
    addTask: (task) => {
        set((state) => ({
            tasks: [task, ...state.tasks],
        }));
    },

    // 알림 추가
    addNotification: (notification) => {
        set((state) => ({
            notifications: [notification, ...state.notifications],
        }));
    },
}));

export default useTaskStore;
```

```jsx
// src/components/TaskBoard.jsx
import React, { useEffect, useState } from 'react';
import { DragDropContext, Droppable, Draggable } from 'react-beautiful-dnd';
import TaskCard from './TaskCard';
import useTaskStore from '../store/taskStore';

const COLUMNS = {
    PENDING: { title: '대기중', color: 'bg-gray-100' },
    IN_PROGRESS: { title: '진행중', color: 'bg-blue-100' },
    COMPLETED: { title: '완료', color: 'bg-green-100' },
};

export default function TaskBoard({ team }) {
    const { tasks, fetchTasks, updateTaskStatus } = useTaskStore();
    const [columns, setColumns] = useState({});

    useEffect(() => {
        fetchTasks(team);
    }, [team, fetchTasks]);

    // 태스크를 상태별로 그룹화
    useEffect(() => {
        const grouped = {
            PENDING: tasks.filter((t) => t.status === 'PENDING'),
            IN_PROGRESS: tasks.filter((t) => t.status === 'IN_PROGRESS'),
            COMPLETED: tasks.filter((t) => t.status === 'COMPLETED'),
        };
        setColumns(grouped);
    }, [tasks]);

    // 드래그 완료 시
    const handleDragEnd = async (result) => {
        const { source, destination, draggableId } = result;

        if (!destination) return;
        if (
            source.droppableId === destination.droppableId &&
            source.index === destination.index
        ) {
            return;
        }

        const taskId = parseInt(draggableId);
        const newStatus = destination.droppableId;

        // API 호출
        await updateTaskStatus(taskId, newStatus);
    };

    return (
        <DragDropContext onDragEnd={handleDragEnd}>
            <div className="flex gap-4 p-6 bg-gray-50 rounded-lg">
                {Object.entries(COLUMNS).map(([status, { title, color }]) => (
                    <Droppable key={status} droppableId={status}>
                        {(provided, snapshot) => (
                            <div
                                ref={provided.innerRef}
                                {...provided.droppableProps}
                                className={`flex-1 rounded-lg p-4 min-h-96 ${color} ${
                                    snapshot.isDraggingOver ? 'bg-blue-200' : ''
                                }`}
                            >
                                <h3 className="font-bold mb-4">
                                    {title} ({columns[status]?.length || 0})
                                </h3>

                                <div className="space-y-3">
                                    {columns[status]?.map((task, index) => (
                                        <Draggable
                                            key={task.id}
                                            draggableId={String(task.id)}
                                            index={index}
                                        >
                                            {(provided, snapshot) => (
                                                <div
                                                    ref={provided.innerRef}
                                                    {...provided.draggableProps}
                                                    {...provided.dragHandleProps}
                                                    className={snapshot.isDragging ? 'opacity-50' : ''}
                                                >
                                                    <TaskCard task={task} />
                                                </div>
                                            )}
                                        </Draggable>
                                    ))}
                                </div>

                                {provided.placeholder}
                            </div>
                        )}
                    </Droppable>
                ))}
            </div>
        </DragDropContext>
    );
}
```

```javascript
 * Slack 메시지 발송
 */
async function sendSlackMessage(channel, text, blocks = null) {
    try {
        const payload = {
            channel,
            text,
        };

        if (blocks) {
            payload.blocks = blocks;
        }

        const response = await slack.post('/chat.postMessage', payload);

        if (!response.data.ok) {
            console.error('Slack error:', response.data.error);
            return null;
        }

        console.log(`✅ Message sent to ${channel}`);
        return response.data;
    } catch (err) {
        console.error('❌ Slack API error:', err.message);
    }
}
```

```sql
-- 각 태스크마다 assigned_name 조회 → 50개 태스크 = 50번 쿼리
```

### 2-2. API 응답 캐싱 (Redis 선택사항)

```javascript
// 간단한 메모리 캐시
const cache = {};

async function getCachedTasks(team) {
    const key = `tasks_${team}`;
    
    // 캐시가 있고 5분 이내면 그것 사용
    if (cache[key] && Date.now() - cache[key].timestamp < 300000) {
        return cache[key].data;
    }

    const data = await db.query(
        'SELECT * FROM tasks WHERE team = $1',
        [team]
    );
    
    cache[key] = { data: data.rows, timestamp: Date.now() };
    return data.rows;
}
```
